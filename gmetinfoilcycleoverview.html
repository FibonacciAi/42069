4. Revisiting T35 Cycles v1 Code

Below is the original code snippet that generates 3D spiral visualizations of a 55-day cycle with key days (13, 28, 33). It uses plotly.graph_objects to render lines and markers.

import numpy as np
import plotly.graph_objects as go

# Parameters for the cycle
phases = 3
cycle_length = 55
key_days = [13, 28, 33]

# Generate spiral data
spiral_data = []
for phase in range(phases):
    for key_day in key_days:
        angle = 2 * np.pi * (key_day + phase * cycle_length) / cycle_length
        x = (key_day + phase * cycle_length) * np.cos(angle)
        y = (key_day + phase * cycle_length) * np.sin(angle)
        z = key_day
        spiral_data.append((x, y, z, phase))

# Create 3D Scatter Plot
fig = go.Figure()

colors = ['blue', 'red', 'green']

for phase in range(phases):
    phase_data = [d for d in spiral_data if d[3] == phase]
    x_vals = [d[0] for d in phase_data]
    y_vals = [d[1] for d in phase_data]
    z_vals = [d[2] for d in phase_data]
    
    fig.add_trace(go.Scatter3d(
        x=x_vals,
        y=y_vals,
        z=z_vals,
        mode='lines+markers',
        line=dict(color=colors[phase], width=4),
        marker=dict(size=5),
        name=f'Phase {phase + 1}'
    ))

# Update layout
fig.update_layout(
    scene=dict(
        xaxis_title='X Axis',
        yaxis_title='Y Axis',
        zaxis_title='Cycle Day'
    ),
    title='Nested Spirals Representation of 55-Day Cycle',
)

# Display the plot
fig.show()

Below is a condensed example that merges a 55-day and a 255-day cycle into a single 3D spiral visualization. We include a phase_shift parameter and multiple key_days for each cycle.

import numpy as np
import plotly.graph_objects as go

def generate_spiral_data(cycle_length, key_days, phases=3, phase_shift=0):
    data = []
    for phase in range(phases):
        for kd in key_days:
            angle = 2 * np.pi * (kd + phase * cycle_length + phase_shift) / cycle_length
            r = (kd + phase * cycle_length + phase_shift)
            x = r * np.cos(angle)
            y = r * np.sin(angle)
            z = kd
            data.append((x, y, z, phase))
    return data

# Primary 55-day cycle
cycle_length_55 = 55
key_days_55 = [13, 28, 33]

# Extended 255-day cycle
cycle_length_255 = 255
key_days_255 = [13, 28, 33, 132]  # Example extended key days

# Generate data for both cycles
spiral_data_55 = generate_spiral_data(cycle_length_55, key_days_55, phases=3, phase_shift=0)
spiral_data_255 = generate_spiral_data(cycle_length_255, key_days_255, phases=2, phase_shift=10)

fig = go.Figure()

# Plot 55-day cycle
colors_55 = ['blue', 'red', 'green']
for phase in range(3):
    phase_data = [d for d in spiral_data_55 if d[3] == phase]
    x_vals = [d[0] for d in phase_data]
    y_vals = [d[1] for d in phase_data]
    z_vals = [d[2] for d in phase_data]
    fig.add_trace(go.Scatter3d(
        x=x_vals, y=y_vals, z=z_vals,
        mode='lines+markers',
        line=dict(color=colors_55[phase], width=4),
        marker=dict(size=5),
        name=f'55d Phase {phase + 1}'
    ))

# Plot 255-day cycle
colors_255 = ['orange', 'purple']
for phase in range(2):
    phase_data = [d for d in spiral_data_255 if d[3] == phase]
    x_vals = [d[0] for d in phase_data]
    y_vals = [d[1] for d in phase_data]
    z_vals = [d[2] for d in phase_data]
    fig.add_trace(go.Scatter3d(
        x=x_vals, y=y_vals, z=z_vals,
        mode='lines+markers',
        line=dict(color=colors_255[phase], width=4),
        marker=dict(size=5),
        name=f'255d Phase {phase + 1}'
    ))

fig.update_layout(
    scene=dict(
        xaxis_title='X Axis',
        yaxis_title='Y Axis',
        zaxis_title='Cycle Day'
    ),
    title='Multi-Cycle Spiral (55-Day & 255-Day)'
)

fig.show()
